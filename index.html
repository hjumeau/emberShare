Qu'est ce qu'Ember?

  • Un Framework Javascript MVC
  • Open source créé en décembre 2011
  • Conçu pour développer des Single-Page-Application

Pourquoi Ember?

  • Ultra complet (tout ce dont vouz auriez besion et là !)
  • Architecture strict pour un code réutilisable et clair
  • Convention plutôt que configuration
  • Hautement productif

Concepts de base

  • Application
  • Run loop
  • Model Object
  • Template
  • MVC*
  • Container

Model

(Ember-Data)

Une couche d’abstraction de stockage de donnés pour simplifier la récupération, la mise à jour et la persistence des données

http://emberjs.com/guides/models/

App.Person = DS.Model.extend({   // person -> App.Person
	firstName: DS.attr('string'),
	lastName: DS.attr('string'),
	
	fullName: function() {
		return this.get('firstName') + ' ' + this.get('lastName');
	}.property('firstName', 'lastName')
});

this.store.createRecord('person',{firstName: 'James Francis', lastName: 'Ryan'}); // POST + json
var ryan = this.store.find('person', 1); // => GET /posts/1

ryan.get(fullName); // James Francis Ryan

Router

Location API => hash / history.pushState / none

/*  /#/posts/new => posts.new route  */	
App.Router.map(function() {
  this.resource('posts', function() {
    this.route('new');
  });
});

/*  /posts/new  */
App.Router.reopen({
  location: 'history'
});

/*  none  */
App.Router.reopen({
  location: 'none'
});

Router

Defininig yours route

http://emberjs.com/guides/routing/defining-your-routes/

App.Router.map(function() {
  this.resource('post', { path: '/post/:post_id' }, function() {
    this.route('edit');
    this.resource('comments', function() {
      this.route('new');
    });
  });
});
URL Route Name Controller Route Template
/ index App.IndexController App.IndexRoute index
N/A post App.PostController App.PostRoute post
/post/:post_id2 post.index App.PostIndexController App.PostIndexRoute post/index
/post/:post_id/edit post.edit App.PostEditController App.PostEditRoute post/edit
N/A comments App.CommentsController App.CommentsRoute comments
/post/:post_id/comments comments.index App.CommentsIndexController App.CommentsIndexRoute comments/index
/post/:post_id/comments/new comments.new App.CommentsNewController App.CommentsNewRoute comments/new

App.PhotoRoute = Ember.Route.extend({
  model: function(params) {
    return Ember.$.getJSON('/photos/'+params.photo_id);
  },
  setupController: function(controller, model) {
    controller.set('model', model);
  },
  renderTemplate: function() {
    this.render('favoritePost', {   // the template to render
      into: 'posts',                // the template to render into
      outlet: 'posts',              // the name of the outlet in that template
      controller: 'blogPost'        // the controller to use for the template
    });
  }
});
